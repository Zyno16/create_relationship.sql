 CREATE TABLE DEPT(
     DEPTNO INT PRIMARY KEY,
     DEPTNAME VARCHAR(50) NOT NULL
 ); 

CREATE TABLE EMP(
     EMPNO INT PRIMARY KEY,
     EMPNAME VARCHAR(50) NOT NULL,
    EMPSALARY DOUBLE ,
    DEPTWORK INT,
    CONSTRAINT FOREIGN KEY(DEPTWORK) REFERENCES DEPT(DEPTNO)
       
     );
